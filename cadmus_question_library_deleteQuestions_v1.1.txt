// console function to delete selected questions in Cadmus library

(async () => {
  // --- 1. Find Apollo Client ---
  const root = document.getElementById('root');
  const fk = Object.keys(root).find(k => k.startsWith('__reactFiber') || k.startsWith('__reactContainer'));

  function bfsFind(startFiber, matcher, maxNodes = 5000) {
    const queue = [startFiber];
    const seen = new WeakSet();
    let n = 0;
    while (queue.length && n < maxNodes) {
      const f = queue.shift();
      if (!f || seen.has(f)) continue;
      seen.add(f); n++;
      const r = matcher(f);
      if (r) return r;
      if (f.child) queue.push(f.child);
      if (f.sibling) queue.push(f.sibling);
    }
    return null;
  }

  const apolloClient = bfsFind(root[fk], f => {
    const v = f.memoizedProps?.value;
    if (v && typeof v === 'object' && v.refetchQueries && v.cache) return v;
    if (v?.client?.refetchQueries && v?.client?.cache) return v.client;
    return null;
  });

  if (!apolloClient) return console.error('Apollo Client not found.');

  // --- 2. Get selected question IDs from TanStack Table ---
  const checkboxes = document.querySelectorAll('input[type="checkbox"][aria-label="Select this row"]');
  if (!checkboxes.length) return console.warn('No questions found in the table.');

  const fiberKey = Object.keys(checkboxes[0]).find(k => k.startsWith('__reactFiber'));
  let tableInstance = null;
  let fiber = checkboxes[0][fiberKey];
  for (let i = 0; i < 30 && fiber; i++, fiber = fiber.return) {
    if (fiber.memoizedProps?.table?.getSelectedRowModel) {
      tableInstance = fiber.memoizedProps.table;
      break;
    }
  }

  if (!tableInstance) return console.error('Could not find table instance.');

  const selectedRows = tableInstance.getSelectedRowModel().rows;
  if (!selectedRows.length) return console.warn('No questions selected.');

  // --- 3. Build request headers from URL ---
  const urlMatch = window.location.href.match(/\/([^/]+)\/assessment\/([^/]+)/);
  if (!urlMatch) return console.error('Could not parse tenant/assessment from URL.');
  const [, tenant, assessmentId] = urlMatch;

  const headers = {
    'Content-Type': 'application/json',
    'x-cadmus-role': 'lecturer',
    'x-cadmus-tenant': tenant,
    'x-cadmus-assessment': assessmentId,
    'x-cadmus-url': window.location.href,
  };

  const mutation = `
    mutation ArchiveQuestion($questionId: ID!) {
      archiveQuestion(questionId: $questionId) { id __typename }
    }
  `;

  // --- 4. Delete each selected question ---
  console.log(`Deleting ${selectedRows.length} question(s)...`);

  for (const row of selectedRows) {
    const { id, shortPrompt } = row.original;
    console.log(`  Deleting: ${shortPrompt?.substring(0, 60)}... (${id})`);

    const res = await fetch('https://api.cadmus.io/cadmus/api/graphql', {
      method: 'POST',
      headers,
      credentials: 'include',
      body: JSON.stringify({ operationName: 'ArchiveQuestion', query: mutation, variables: { questionId: id } }),
    });

    const json = await res.json();
    if (json.errors) console.error(`  ✗ Failed:`, json.errors);
    else console.log(`  ✓ Deleted`);
  }

  // --- 5. Refresh the list in-place ---
  await apolloClient.refetchQueries({ include: ['LibraryQuestions'] });
  console.log('Done. List refreshed.');
})();
